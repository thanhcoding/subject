#include<stdio.h>

void out(int a[],int n);
void Try(int a[],int n,int i);
int d=0;
int main(){
	int n;
	printf("enter n: ");
	scanf("%d",&n);
	int a[n+1];
	Try(a,n,1);
	return 0;
}

void out(int a[],int n){
	printf("%d: ",++d);
	for(int i=1;i<=n;i++){
		printf("%d ",a[i]);
	}
	printf("\n");
}
void Try(int a[],int n,int i){
	for(int j=0;j<=1;j++){
		if(j==0 || j==1){
			a[i]=j;
		
		if(i==n){
			out(a,n);
		}
		else Try(a,n,i+1);
		}
	}
}
